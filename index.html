    document.getElementById('songPage').classList.add('active');

    header.innerText = songs[index].title;

    document.getElementById('songTitlePage').innerText = songs[index].title;

    document.getElementById('songLyrics').innerText = songs[index].lyrics;

    currentFontSize = 12;

    document.getElementById('songLyrics').style.fontSize = currentFontSize + 'px';

  }

 

  function openAbout() {

    document.getElementById('menuPage').classList.remove('active');

    document.getElementById('aboutPage').classList.add('active');

    header.innerText = 'Waa’ee keenya';

  }

 

  function goBack() {

    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));

    document.getElementById('menuPage').classList.add('active');

    header.innerText = 'Baafata';

  }

 

  let currentFontSize = 12;

  function zoomIn() {

    currentFontSize += 2;

    document.getElementById('songLyrics').style.fontSize = currentFontSize + 'px';

  }

 

  function zoomOut() {

    if (currentFontSize > 6) {

      currentFontSize -= 2;

      document.getElementById('songLyrics').style.fontSize = currentFontSize + 'px';

    }

  }
